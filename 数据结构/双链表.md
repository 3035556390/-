双链表

结点定义

```c++
typedef struct Dnode {
    int data;
    struct Dnode* prior;
    struct Dnode* next;
}DLinkNode ;
```

头插法

```c++
void CreateListF(DLinkNode*& L, int a[], int n) {
    L = (DLinkNode*)malloc(sizeof(DLinkNode));
    L->next = NULL;
    L->prior = NULL;
    for (int i = 0; i < n; i++)
    {
        DLinkNode*s= (DLinkNode*)malloc(sizeof(DLinkNode));
        s->data = a[i];
        s->next = L->next;
        if (L->next!=NULL)
        {
            L->next->prior = s;
        }
        L->next = s;
        s->prior = L;
    }
}
```

尾插法

```c++
void CreateListR(DLinkNode*& L, int a[], int n) {
    L = (DLinkNode*)malloc(sizeof(DLinkNode));
    L->next = NULL;
    L->prior = NULL;
    DLinkNode* r = L;
    for (int i = 0; i < n; i++)
    {
        DLinkNode* s = (DLinkNode*)malloc(sizeof(DLinkNode));
        s->data = a[i];
        r->next = s;
        s->prior = r;
        r = s;
    }
    r->next = NULL;
}
```
