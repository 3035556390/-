单链表

结点

```c++
typedef struct node{
    int data;
     struct node* next;
} LinkNode;
```

头插法(逆顺序插入)

```c++
void CreateListF(LinkNode*& L, int a[], int n) {
    L = (LinkNode*)malloc(sizeof(LinkNode));
    L->next = NULL;
    for (int i = 0; i < n; i++)
    {LinkNode*s= (LinkNode*)malloc(sizeof(LinkNode));
    s->data = a[i];
   
    s->next = L->next;
    L->next = s;
    }
}
```

尾插法

```c++
void CreateListR(LinkNode*& L, int a[], int n) {
    LinkNode* r;
   r=L = (LinkNode*)malloc(sizeof(LinkNode));
    L->next = NULL;
    for (int i = 0; i < n; i++)
    {
        LinkNode* s = (LinkNode*)malloc(sizeof(LinkNode));
        s->data = a[i];
        r->next = s;
        r = s;
    }
    r->next = NULL;
}
```

销毁单链表

```c++
void DestroyList(LinkNode*&L) {
    LinkNode* pre, * p;
    pre = L;
    p = pre->next;
    while (p != NULL) {
        free(pre);
        pre = p;
        p = p->next;
    }
    free(pre);
}
```

求单链表长度

```c++
int ListLength(LinkNode*& L) {
    LinkNode* p = L;
    int n = 0;
    while (p->next != NULL) {
        n++;
        p = p->next;
    }
    return n;

}
```

获取第i个元素

```c++
bool GetElem(LinkNode*& L, int i, int& e) {
    int j = 1;
    LinkNode* p = L->next;
    while (p != NULL && j < i) {
        j++;
        p = p->next;
    }
    if (p==NULL)
    {
        return false;
    }
    else {
        e = p->data;
        return true;
    }

}

```

```c++
int LocateElem(LinkNode*& L, int elem) {
    int i = 1;
    LinkNode* p = L->next;

    while (p != NULL && p->data != elem) {
        i++;
        p = p->next;
    }
    if (p==NULL)
    {
        return 0;
    }
    else { return i;
    }
}
```

```c++
bool ListDelete(LinkNode*& L, int i,int &e) {
    LinkNode* p = L;
    int j = 0;
    while (p != NULL && j < i - 1) {
        j++;
        p = p->next;
    }
    if (p==NULL)
    {
        return false;
    }
    else {
        if (p->next!=NULL)
        {
            LinkNode* q = p->next;
            p->next = q->next;
            e = q->data;
            free(q);
        }
    }
}
```
