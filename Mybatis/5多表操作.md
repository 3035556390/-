一对一association     本质是将一条查询结果对应一个实体对象

```java
public class Card {

    private Integer id;
    private String number;
    private Person person;//在实体对象里引用另一个对象达到一对一
}
```



```java
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="org.example.springbootmybatis.one_to_one.Interface.one_to_oneMapperInterface">
    //实体对象属性<-> property <-> column <->查询结果
    //说白了就是把column列查询结果存放到对象属性property中
    
//resultMap标签来定义结果类型
    
    <resultMap id="card" type="org.example.springbootmybatis.bean.Card">
   //配置本身对象的映射关系 
    //property属性值必须和映射对象的属性名一样    column属性值必须和sql语句的查询别名一致
        <id property="id" column="cid"/>  //用来指定id
        <result property="number" column="number"/> //用来指定其它属性

        <association property="person"   javaType="org.example.springbootmybatis.bean.Person">  //配置关联对象的映射关系
            <id property="id" column="pid"/>
            <result property="name" column="name"/>
            <result property="age" column="age"/>
        </association>

    </resultMap>

    <select id="selectall" resultMap="card" >
        select c.id cid,number,p.id pid,name,age from card c,person p where c.pid=p.id
    </select>
</mapper>

```

```java
public class Student {
    private Integer sno;
    private String sname;
    private String ssex;
    private String smajor;

    private List<Course> courseList;//引用list达到一对多,多对多
}
```

```java
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="org.example.springbootmybatis.one_to_many.Interface.one_to_manyMapperInterface">


    <resultMap id="Student" type="org.example.springbootmybatis.bean.Student">
        <id property="sno" column="sno"/>
        <result property="sname" column="sname"/>
        <result property="ssex" column="sex"/>
        <result property="smajor" column="smajor"/>


        <collection property="courseList" ofType="org.example.springbootmybatis.bean.Course">
            <id property="cno" column="cno"/>
            <result property="cname" column="cname"/>
        </collection>

    </resultMap>

    <select id="selectall" resultMap="Student" >
        SELECT s.sno,s.Sname,s.Sex,s.Smajor,c.cno,Cname FROM student s,course c,sc WHERE s.sno=sc.Sno and sc.Cno=c.cno
        </select>
</mapper>
```
